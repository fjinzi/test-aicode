# Redis初始化命令
# 执行方式：redis-cli < init_redis.txt

# 设置商品信息
SET seckill:product:1 '{"id":1,"name":"iPhone 15 Pro","price":5999,"stock":1000}'

# 设置商品库存
SET seckill:stock:1 1000

# 设置秒杀开始时间（当前时间）
SET seckill:start_time:1 $(date +"%Y-%m-%d %H:%M:%S")

# 设置秒杀结束时间（30分钟后）
SET seckill:end_time:1 $(date -d '+30 minutes' +"%Y-%m-%d %H:%M:%S")
